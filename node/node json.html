将文件中的json解析出来

使用 JSON.parse 将文件里面的json字符串解析出来, 并且放到 map 当中.

// 导入模块
const fs             = require('fs')
var jsonFilePath = '/Users/user/Desktop/temp/temp/imageNames.json'
var data         = fs.readFileSync(jsonFilePath, 'utf-8')
let map = new Map();
var imageJson    = JSON.parse(data, (key, value) => {
  dlog(key, new Info(`key`), 0)
  dlog(value, new Info(`value`), 0)
  map.set(key, value)
})
dlog(map, new Info(`map`), 0)
