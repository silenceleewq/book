iOS 日历添加事件 Calendar

iOS 日历添加事件 Calendar

最近工作中用到了日历添加事件的功能, 没有过多的研究,很多东西也都是从网上博客看到的, 这里也是做一下记录.

说一下思路:

  1. 需要在 info.plist 文件中添加 key NSCalendarsUsageDescription, 没有这个的话, 是不会被审核通过的, 也会崩溃.
  2. 判断权限, 如果用户没有打开, 那么跳转到设置页面, 让用户打开权限.

代码:

// 声明一个属性
@property (nonatomic, strong) EKEventStore *eventStore;

//在初始化方法中初始化属性 eventStore
- (instancetype)init {
  self = [super init];
  if (self) {
    self.eventStore = [EKEventStore new];
  }
  return self;
}


// 需要导入头文件 #import <EventKit/EventKit.h>
- (void)calendarEvent {
  EKAuthorizationStatus eventStatus = [EKEventStore  authorizationStatusForEntityType:EKEntityTypeEvent];

  // 判断权限状态
  if (eventStatus == EKAuthorizationStatusDenied || eventStatus == EKAuthorizationStatusRestricted ) {
    NSLog(@"日历权限不可用");
    // [[UIApplication sharedApplication]openURL:[NSURL URLWithString:UIApplicationOpenSettingsURLString]];
    [[UIApplication sharedApplication] openURL:[NSURL URLWithString:UIApplicationOpenSettingsURLString] options:@{} completionHandler:nil];
  } else {
    NSLog(@"日历权限可用");
    [self addEventToCalendar];
  }
}

- (void)addEventToCalendar {
  // 请求添加事件
  [self.eventStore requestAccessToEntityType:EKEntityTypeEvent completion:^(BOOL granted, NSError *_Nullable error) {
    if (error) {
      NSLog(@"error %@", error.localizedDescription);
    }
    if (!granted) {
      return;
    }

    EKEvent *event = [EKEvent eventWithEventStore:self.eventStore];
    // 事件标题
    event.title = @"hello";

    // 初始化一个日期格式化对象. 这个目的是计算出事件的起始事件. 当然,事件的起始时间也可以设置为同一个时间, 比如起始时间都是上午11点.
    NSDateFormatter *dateFormatter = [[NSDateFormatter alloc] init];
    [dateFormatter setAMSymbol:@"AM"];
    [dateFormatter setPMSymbol:@"PM"];
    NSString *startDateString = @"2019/11/19 10:30:00";
    NSString *endDateString = @"2019/11/19 18:30:00";
    NSString *alarmDateString = @"2019/11/19 11:10:00";
    [dateFormatter setDateFormat:@"yyyy/MM/dd HH:mm:ss"];
    NSDate *startDate = [dateFormatter dateFromString:startDateString];
    NSDate *endDate = [dateFormatter dateFromString:endDateString];
    NSDate *alarmDate = [dateFormatter dateFromString:alarmDateString];

    // 设置事件的
    event.startDate = startDate;
    event.endDate   = endDate;
    // 设置 calendar 源, 这里就是一个简单的默认源, 还有iCloud源好像, 可以网上搜一下.
    [event setCalendar:[self.eventStore defaultCalendarForNewEvents]];

    // 设置这个日期的提醒时间.
    EKAlarm *alarm = [EKAlarm alarmWithAbsoluteDate:alarmDate];
    event.alarms = @[alarm];

    NSError *err;
    // 写入事件.
    [self.eventStore saveEvent:event span:EKSpanThisEvent commit:YES error:&err];
    if (!err) {
      NSLog(@"保存成功!");
    }
  }];
}
