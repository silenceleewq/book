git 远程仓库

git

查看远程仓库地址

git remote -v

添加远程仓库地址

git remote add origin github.git 移除远程仓库

git remote remove origin

设置默认远程分支

添加远程仓库

git remote add -f origin git@github.com:silenceleewq/book.git

打开配置文件:

git config -e

[core]
     repositoryformatversion = 0
     filemode = true
     bare = false
     logallrefupdates = true
     ignorecase = true
     precomposeunicode = true
 [branch "master"]
     remote = origin
     merge = refs/heads/master
 [remote "origin"]

添加 下面内容即可:

[branch "master"]
    remote = origin
    merge = refs/heads/master
